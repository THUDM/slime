---
  Slime 多机多卡训练指南

  核心架构

  Slime 使用 Ray 作为分布式调度框架，通过 Ray 的 placement group 来管理多机多卡资源分配。

  关键参数

  # 训练资源分配
  --actor-num-nodes 2              # Actor 训练使用的节点数
  --actor-num-gpus-per-node 8      # 每个节点的 GPU 数

  # 推理资源分配
  --rollout-num-gpus 16            # Rollout 推理使用的总 GPU 数
  --rollout-num-gpus-per-engine 8  # 每个推理引擎的 GPU 数（类似 tp_size）

  # 训推一体模式
  --colocate                       # 训练和推理共用 GPU

  多机启动流程

  步骤 1：在主节点启动 Ray Head
  export MASTER_ADDR=<主节点IP>
  ray start --head --node-ip-address ${MASTER_ADDR} --num-gpus 8 --disable-usage-stats

  步骤 2：在其他节点加入 Ray 集群
  ray start --address=${MASTER_ADDR}:6379 --num-gpus 8 --node-ip-address ${WORKER_IP} --disable-usage-stats

  步骤 3：提交训练任务
  ray job submit --address="http://127.0.0.1:8265" \
     --runtime-env-json='{
       "env_vars": {
          "MASTER_ADDR": "${MASTER_ADDR}",
          "PYTHONPATH": "/root/Megatron-LM/",
          ...
       }
     }' \
     -- python3 train.py \
     --actor-num-nodes 2 \
     --actor-num-gpus-per-node 8 \
     --colocate \
     ...

  自动化多机启动脚本示例

  如果有 hostfile（每行格式：ip slot=8），可以自动启动所有节点：

  # 启动 head 节点
  ray start --head --node-ip-address ${MASTER_ADDR} --num-gpus 8 --disable-usage-stats

  # 自动启动 worker 节点
  for WORKER_IP in $(awk '{print $1}' /path/to/hostfile); do
    if [[ "$WORKER_IP" == "$MASTER_ADDR" ]]; then
      continue
    fi
    echo "Starting Ray worker on ${WORKER_IP}"
    ssh root@"${WORKER_IP}" \
      "pkill -9 sglang ; ray stop --force ; ray start --address=${MASTER_ADDR}:6379 --num-gpus 8 --node-ip-address ${WORKER_IP} --disable-usage-stats" &
  done
  wait

  资源分配示例

  2 机 16 卡训推一体：
  --actor-num-nodes 2
  --actor-num-gpus-per-node 8
  --colocate

  4 机 32 卡训推分离：
  --actor-num-nodes 2
  --actor-num-gpus-per-node 8
  --rollout-num-gpus 16

  8 机 64 卡大模型训练（如 GLM-4.5 355B）：
  --actor-num-nodes 8
  --actor-num-gpus-per-node 8
  --colocate
  --tensor-model-parallel-size 8
  --pipeline-model-parallel-size 4
  --context-parallel-size 2

  重要环境变量

  export MASTER_ADDR=<主节点IP>
  export PYTHONPATH=/root/Megatron-LM/
  export CUDA_DEVICE_MAX_CONNECTIONS=1
  export NCCL_NVLS_ENABLE=0  # 根据是否有 NVLink 设置

  你的 KIE 训练多机配置示例

  # 假设 2 机 16 卡
  export MASTER_ADDR=<node0_ip>

  # Node 0
  ray start --head --node-ip-address ${MASTER_ADDR} --num-gpus 8 --disable-usage-stats

  # Node 1 (SSH 或手动执行)
  ray start --address=${MASTER_ADDR}:6379 --num-gpus 8 --node-ip-address <node1_ip> --disable-usage-stats

  # 提交训练
  ray job submit --address="http://127.0.0.1:8265" \
     --runtime-env-json='{"env_vars": {"MASTER_ADDR": "'${MASTER_ADDR}'"}}' \
     -- python3 train.py \
     --hf-checkpoint /path/to/model \
     --prompt-data /path/to/kie_rl_train.parquet \
     --actor-num-nodes 2 \
     --actor-num-gpus-per-node 8 \
     --colocate \
     --train-backend fsdp \
     ...
